# 有道Agent实施计划

> **智能体执行指南：** 必须使用子技能：superpowers:subagent-driven-development（推荐）或 superpowers:executing-plans 逐任务执行本计划。

**目标：** 实现有道云笔记绑定/解绑、笔记浏览、预览和批量导入功能。

**架构：** YoudaoAgentService 管理有道绑定状态，通过 YoudaoNoteSkill（外部依赖）访问有道API。绑定信息加密存储在 youdao_binding 表。

**技术栈：** Gin、GORM

```go
// internal/service/youdao_agent_interface.go
package service

import (
	"YoudaoNoteLm/internal/model/dto/response"
	"YoudaoNoteLm/internal/model/entity"
)

// YoudaoAgentService 有道云笔记Agent服务接口
type YoudaoAgentService interface {
	Bind(userID uint, apiKey string) error
	Unbind(userID uint) error
	GetStatus(userID uint) (*response.BindingStatusResponse, error)
	ListNotes(userID uint, page, size int, keyword string) (*NoteListResponse, error)
	PreviewNote(userID uint, noteID string) (string, error)
	ImportNotes(userID, notebookID uint, noteIDs []string) (*entity.ImportTask, error)
}

// NoteListResponse 笔记列表响应
type NoteListResponse struct {
	Notes []*response.YoudaoNoteResponse `json:"notes"`
	Total int64                           `json:"total"`
}

// YoudaoNoteSkill 有道云笔记技能接口（外部依赖）
type YoudaoNoteSkill interface {
	ListNotes(apiKey string, page, size int, keyword string) ([]*response.YoudaoNoteResponse, int64, error)
	GetNoteContent(apiKey string, noteID string) (string, error)
}
```

- [ ] **步骤 2：创建 YoudaoAgentService 实现**

```go
// internal/service/youdao_agent_service.go
package service

import (
	"YoudaoNoteLm/internal/model/dto/response"
	"YoudaoNoteLm/internal/model/entity"
	"YoudaoNoteLm/internal/repository"
	bizerrors "YoudaoNoteLm/pkg/errors"
	"YoudaoNoteLm/pkg/logger"

	"go.uber.org/zap"
)

type youdaoAgentService struct {
	youdaoSkill YoudaoNoteSkill
	bindingRepo repository.YoudaoBindingRepository
	importer    ImporterService
}

func NewYoudaoAgentService(
	youdaoSkill YoudaoNoteSkill,
	bindingRepo repository.YoudaoBindingRepository,
	importer ImporterService,
) YoudaoAgentService {
	return &youdaoAgentService{
		youdaoSkill: youdaoSkill,
		bindingRepo: bindingRepo,
		importer:    importer,
	}
}

func (s *youdaoAgentService) Bind(userID uint, apiKey string) error {
	existing, _ := s.bindingRepo.FindByUserID(userID)
	if existing != nil && existing.Status == "active" {
		return bizerrors.New(bizerrors.CodeConflict, "已绑定有道云笔记，请先解绑")
	}

	binding := &entity.YoudaoBinding{
		UserID: userID,
		APIKey: apiKey,
		Status: "active",
	}

	if existing != nil {
		existing.APIKey = apiKey
		existing.Status = "active"
		return s.bindingRepo.Update(existing)
	}

	return s.bindingRepo.Create(binding)
}

func (s *youdaoAgentService) Unbind(userID uint) error {
	return s.bindingRepo.Delete(userID)
}

func (s *youdaoAgentService) GetStatus(userID uint) (*response.BindingStatusResponse, error) {
	binding, err := s.bindingRepo.FindByUserID(userID)
	if err != nil {
		return nil, err
	}

	if binding == nil {
		return &response.BindingStatusResponse{Bound: false, Status: ""}, nil
	}

	return &response.BindingStatusResponse{
		Bound:  binding.Status == "active",
		Status: binding.Status,
	}, nil
}

func (s *youdaoAgentService) ListNotes(userID uint, page, size int, keyword string) (*NoteListResponse, error) {
	binding, err := s.bindingRepo.FindByUserID(userID)
	if err != nil {
		return nil, err
	}
	if binding == nil || binding.Status != "active" {
		return nil, bizerrors.New(bizerrors.CodeInvalidYoudaoAPIKey, "未绑定有道云笔记")
	}

	notes, total, err := s.youdaoSkill.ListNotes(binding.APIKey, page, size, keyword)
	if err != nil {
		return nil, err
	}

	return &NoteListResponse{Notes: notes, Total: total}, nil
}

func (s *youdaoAgentService) PreviewNote(userID uint, noteID string) (string, error) {
	binding, err := s.bindingRepo.FindByUserID(userID)
	if err != nil {
		return "", err
	}
	if binding == nil || binding.Status != "active" {
		return "", bizerrors.New(bizerrors.CodeInvalidYoudaoAPIKey, "未绑定有道云笔记")
	}

	return s.youdaoSkill.GetNoteContent(binding.APIKey, noteID)
}

func (s *youdaoAgentService) ImportNotes(userID, notebookID uint, noteIDs []string) (*entity.ImportTask, error) {
	binding, err := s.bindingRepo.FindByUserID(userID)
	if err != nil {
		return nil, err
	}
	if binding == nil || binding.Status != "active" {
		return nil, bizerrors.New(bizerrors.CodeInvalidYoudaoAPIKey, "未绑定有道云笔记")
	}

	task := &entity.ImportTask{
		UserID: userID, NotebookID: notebookID, TaskType: "youdao",
		TotalCount: len(noteIDs), Status: "running",
	}

	logger.Info("有道笔记批量导入开始",
		zap.Uint("user_id", userID),
		zap.Int("count", len(noteIDs)),
	)

	return task, nil
}
```

```go
// internal/api/v1/youdao/controller.go
package youdao

import (
	"YoudaoNoteLm/internal/middleware"
	"YoudaoNoteLm/internal/model/dto/request"
	"YoudaoNoteLm/internal/service"
	"YoudaoNoteLm/pkg/response"
	"strconv"

	"github.com/gin-gonic/gin"
)

type Controller struct {
	youdaoService service.YoudaoAgentService
}

func NewController(youdaoService service.YoudaoAgentService) *Controller {
	return &Controller{youdaoService: youdaoService}
}

func (ctrl *Controller) Bind(c *gin.Context) {
	userID := middleware.GetUserID(c)

	var req request.YoudaoBindRequest
	if err := c.ShouldBindJSON(&req); err != nil {
		response.BadRequest(c, err.Error())
		return
	}

	if err := ctrl.youdaoService.Bind(userID, req.APIKey); err != nil {
		response.BizError(c, err)
		return
	}

	response.SuccessWithMessage(c, "绑定成功", nil)
}

func (ctrl *Controller) Unbind(c *gin.Context) {
	userID := middleware.GetUserID(c)

	if err := ctrl.youdaoService.Unbind(userID); err != nil {
		response.BizError(c, err)
		return
	}

	response.SuccessWithMessage(c, "解绑成功", nil)
}

func (ctrl *Controller) GetStatus(c *gin.Context) {
	userID := middleware.GetUserID(c)

	status, err := ctrl.youdaoService.GetStatus(userID)
	if err != nil {
		response.BizError(c, err)
		return
	}

	response.Success(c, status)
}

func (ctrl *Controller) ListNotes(c *gin.Context) {
	userID := middleware.GetUserID(c)
	page, _ := strconv.Atoi(c.DefaultQuery("page", "1"))
	size, _ := strconv.Atoi(c.DefaultQuery("size", "10"))
	keyword := c.Query("keyword")

	result, err := ctrl.youdaoService.ListNotes(userID, page, size, keyword)
	if err != nil {
		response.BizError(c, err)
		return
	}

	response.Success(c, result)
}

func (ctrl *Controller) PreviewNote(c *gin.Context) {
	userID := middleware.GetUserID(c)

	var req request.YoudaoNotePreviewRequest
	if err := c.ShouldBindJSON(&req); err != nil {
		response.BadRequest(c, err.Error())
		return
	}

	content, err := ctrl.youdaoService.PreviewNote(userID, req.NoteID)
	if err != nil {
		response.BizError(c, err)
		return
	}

	response.Success(c, gin.H{"content": content})
}

func (ctrl *Controller) ImportNotes(c *gin.Context) {
	userID := middleware.GetUserID(c)
	nbID, err := strconv.ParseUint(c.Param("nbId"), 10, 32)
	if err != nil {
		response.BadRequest(c, "无效的笔记本ID")
		return
	}

	var req request.YoudaoImportRequest
	if err := c.ShouldBindJSON(&req); err != nil {
		response.BadRequest(c, err.Error())
		return
	}

	task, err := ctrl.youdaoService.ImportNotes(userID, uint(nbID), req.NoteIDs)
	if err != nil {
		response.BizError(c, err)
		return
	}

	response.Success(c, task)
}
```

- [ ] **步骤 2：创建路由**

```go
// internal/api/v1/youdao/routes.go
package youdao

import "github.com/gin-gonic/gin"

func (ctrl *Controller) RegisterRoutes(r *gin.RouterGroup) {
	youdao := r.Group("/youdao")
	{
		youdao.POST("/bind", ctrl.Bind)
		youdao.DELETE("/unbind", ctrl.Unbind)
		youdao.GET("/status", ctrl.GetStatus)
		youdao.GET("/notes", ctrl.ListNotes)
		youdao.POST("/notes/preview", ctrl.PreviewNote)
	}

	notebooks := r.Group("/notebooks/:nbId/youdao")
	{
		notebooks.POST("/import", ctrl.ImportNotes)
	}
}
```
